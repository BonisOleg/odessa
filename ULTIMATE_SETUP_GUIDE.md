```bash
#!/bin/bash

echo "========================================="
echo "üîç HTML Custom Rules Check"
echo "========================================="

ERROR_COUNT=0
WARNING_COUNT=0

# –ó–Ω–∞—Ö–æ–¥–∏–º–æ –≤—Å—ñ HTML —Ñ–∞–π–ª–∏ (–∫—Ä—ñ–º normalize.css —Ç–∞ node_modules)

if [ -z "$HTML_FILES" ]; then
  echo "‚ö†Ô∏è  No HTML files found in templates/"
  exit 0
fi

# –ü—Ä–∞–≤–∏–ª–æ 1: viewport meta –º–∞—î –º—ñ—Å—Ç–∏—Ç–∏ viewport-fit=cover —Ç–∞ interactive-widget=resizes-content

echo "üì± [Rule 1] Checking viewport meta attributes..."
VIEWPORT_ISSUES=$(echo "$HTML_FILES" | xargs grep -l 'name="viewport"' | while read -r file; do
  if ! grep -q 'viewport-fit=cover' "$file" || ! grep -q 'interactive-widget=resizes-content' "$file"; then
    echo "$file"
  fi
done)

if [ -n "$VIEWPORT_ISSUES" ]; then
  echo "‚ùå Viewport meta tags missing required attributes:"
  echo "$VIEWPORT_ISSUES" | while read -r file; do
    echo "   $file"
  done
  echo "   Required: <meta name=\"viewport\" content=\"width=device-width, initial-scale=1.0, viewport-fit=cover, interactive-widget=resizes-content\">"
  ((ERROR_COUNT++))
else
  echo "‚úÖ All viewport meta tags are correct"
fi

# –ü—Ä–∞–≤–∏–ª–æ 2: inline style="" –∑–∞–±–æ—Ä–æ–Ω–µ–Ω—ñ (–¥—É–±–ª—é—î HTMLHint, –∞–ª–µ –¥–ª—è –Ω–∞–¥—ñ–π–Ω–æ—Å—Ç—ñ)

if [ -n "$INLINE_STYLES" ]; then
  echo "‚ùå Inline styles found (forbidden):"
  echo "$INLINE_STYLES"
  ((ERROR_COUNT++))
else
  echo "‚úÖ No inline styles detected"
fi

# –ü—Ä–∞–≤–∏–ª–æ 3: inline onclick="", onerror="" —Ç–∞ —ñ–Ω—à—ñ event handlers –∑–∞–±–æ—Ä–æ–Ω–µ–Ω—ñ

if [ -n "$INLINE_HANDLERS" ]; then
  echo "‚ùå Inline event handlers found (forbidden):"
  echo "$INLINE_HANDLERS"
  ((ERROR_COUNT++))
else
  echo "‚úÖ No inline event handlers detected"
fi

# –ü—Ä–∞–≤–∏–ª–æ 4: input type="tel" –∞–±–æ type="number" –º–∞—î –º–∞—Ç–∏ inputmode="tel" –∞–±–æ inputmode="decimal"
echo ""
echo "üìû [Rule 4] Checking inputmode for tel/number inputs..."
TEL_INPUTS=$(echo "$HTML_FILES" | xargs grep -n 'type="tel"' | grep -v 'inputmode="tel"' || echo "")
NUMBER_INPUTS=$(echo "$HTML_FILES" | xargs grep -n 'type="number"' | grep -v 'inputmode=' || echo "")

if [ -n "$TEL_INPUTS" ]; then
  echo "‚ö†Ô∏è  Inputs with type=\"tel\" missing inputmode=\"tel\":"
  echo "$TEL_INPUTS"
  ((WARNING_COUNT++))
fi

if [ -n "$NUMBER_INPUTS" ]; then
  echo "‚ö†Ô∏è  Inputs with type=\"number\" missing inputmode (recommend inputmode=\"decimal\"):"
  echo "$NUMBER_INPUTS"
  ((WARNING_COUNT++))
fi

if [ -z "$TEL_INPUTS" ] && [ -z "$NUMBER_INPUTS" ]; then
  echo "‚úÖ All tel/number inputs have correct inputmode"
fi

# –ü—Ä–∞–≤–∏–ª–æ 5: <video> —Ç–µ–≥–∏ –º–∞—é—Ç—å –º—ñ—Å—Ç–∏—Ç–∏ poster, playsinline, muted
echo ""
echo "üé¨ [Rule 5] Checking video tags..."
VIDEO_TAGS=$(echo "$HTML_FILES" | xargs grep -n '<video' || echo "")
if [ -n "$VIDEO_TAGS" ]; then
  echo "$VIDEO_TAGS" | while read -r line; do
    file=$(echo "$line" | cut -d: -f1)
    linenum=$(echo "$line" | cut -d: -f2)
    content=$(echo "$line" | cut -d: -f3-)
    
    issues=""
    echo "$content" | grep -q 'poster=' || issues="${issues}poster "
    echo "$content" | grep -q 'playsinline' || issues="${issues}playsinline "
    echo "$content" | grep -q 'muted' || issues="${issues}muted "
    
    if [ -n "$issues" ]; then
      echo "‚ö†Ô∏è  $file:$linenum missing attributes: $issues"
      ((WARNING_COUNT++))
    fi
  done
else
  echo "‚úÖ No video tags found (or all are correct)"
fi

# –ü—Ä–∞–≤–∏–ª–æ 6: <script> —Ç–µ–≥–∏ –º–∞—é—Ç—å –º—ñ—Å—Ç–∏—Ç–∏ defer –∞–±–æ async
echo ""
echo "üìú [Rule 6] Checking script tags for defer/async..."
SCRIPT_TAGS=$(echo "$HTML_FILES" | xargs grep -n '<script src=' | grep -v 'defer\|async' || echo "")
if [ -n "$SCRIPT_TAGS" ]; then
  echo "‚ö†Ô∏è  Script tags without defer/async found:"
  echo "$SCRIPT_TAGS"
  ((WARNING_COUNT++))
else
  echo "‚úÖ All external scripts have defer/async"
fi

# –ü—Ä–∞–≤–∏–ª–æ 7: touch-action: manipulation –¥–ª—è —ñ–Ω—Ç–µ—Ä–∞–∫—Ç–∏–≤–Ω–∏—Ö –µ–ª–µ–º–µ–Ω—Ç—ñ–≤ (–ø–µ—Ä–µ–≤—ñ—Ä—è—î–º–æ –≤ CSS, –∞–ª–µ –Ω–∞–≥–∞–¥—É—î–º–æ —Ç—É—Ç)
echo ""
echo "üëÜ [Rule 7] Reminder: Use touch-action: manipulation for interactive elements"
echo "   (This is checked in CSS, ensure buttons/links have this property)"

# –ü—ñ–¥—Å—É–º–æ–∫
echo ""
echo "========================================="
echo "üìä HTML Rules Summary"
echo "========================================="
echo "Errors: $ERROR_COUNT"
echo "Warnings: $WARNING_COUNT"

if [ $ERROR_COUNT -gt 0 ]; then
  echo "‚ùå HTML custom rules check FAILED"
  exit 1
else
  echo "‚úÖ HTML custom rules check PASSED"
  exit 0
fi
```

### 6.2. `scripts/check-css-rules.sh`

```bash
#!/bin/bash
set -e

echo "========================================="
echo "üé® CSS Custom Rules Check"
echo "========================================="

ERROR_COUNT=0
WARNING_COUNT=0

# –ó–Ω–∞—Ö–æ–¥–∏–º–æ –≤—Å—ñ CSS —Ñ–∞–π–ª–∏ (–∫—Ä—ñ–º normalize.css)

  echo "‚ö†Ô∏è  No CSS files found in static/css/"
  exit 0
fi

# –ü—Ä–∞–≤–∏–ª–æ 1: 100vh –º–∞—î –º–∞—Ç–∏ fallback 100dvh (–∞–±–æ –∫–æ–º–µ–Ω—Ç–∞—Ä "Fallback")

  grep -n '100vh' "$file" | while IFS=: read -r linenum line; do
    # –ü–µ—Ä–µ–≤—ñ—Ä—è—î–º–æ, —á–∏ —î 100dvh –∞–±–æ –∫–æ–º–µ–Ω—Ç–∞—Ä "Fallback" –≤ –Ω–∞—Å—Ç—É–ø–Ω–∏—Ö 2 —Ä—è–¥–∫–∞—Ö
    context=$(sed -n "$((linenum-1)),$((linenum+2))p" "$file")
    if ! echo "$context" | grep -qE '100dvh|Fallback'; then
      echo "$file:$linenum: $line"
    fi
  done
done)

if [ -n "$VH_ISSUES" ]; then
  echo "‚ùå Found 100vh without 100dvh fallback:"
  echo "$VH_ISSUES"
  echo "   Fix: Use 'height: 100vh; /* Fallback */ height: 100dvh;'"
  ((ERROR_COUNT++))
else
  echo "‚úÖ All 100vh declarations have fallback"
fi

# –ü—Ä–∞–≤–∏–ª–æ 2: safe-area-inset-* –º–∞—î –≤–∏–∫–æ—Ä–∏—Å—Ç–æ–≤—É–≤–∞—Ç–∏—Å—å –¥–ª—è padding/margin
echo ""
echo "üì± [Rule 2] Checking safe-area-inset usage..."
SAFE_AREA_USAGE=$(echo "$CSS_FILES" | xargs grep -c 'env(safe-area-inset-' | grep -v ':0$' || echo "")
if [ -z "$SAFE_AREA_USAGE" ]; then
  echo "‚ö†Ô∏è  No safe-area-inset usage detected (may be intentional)"
  echo "   Recommendation: Use env(safe-area-inset-bottom) for fixed elements"
  ((WARNING_COUNT++))
else
  echo "‚úÖ safe-area-inset is used: $(echo "$SAFE_AREA_USAGE" | wc -l) file(s)"
fi

# –ü—Ä–∞–≤–∏–ª–æ 3: font-size –º–∞—î –±—É—Ç–∏ –≤ rem, –∞ –Ω–µ px (warning, –Ω–µ error)
echo ""
echo "üî§ [Rule 3] Checking font-size units (prefer rem over px)..."
PX_FONT_SIZES=$(echo "$CSS_FILES" | xargs grep -n 'font-size:.*px' || echo "")
if [ -n "$PX_FONT_SIZES" ]; then
  echo "‚ö†Ô∏è  font-size in px found (recommend rem for accessibility):"
  echo "$PX_FONT_SIZES" | head -n 10
  if [ $(echo "$PX_FONT_SIZES" | wc -l) -gt 10 ]; then
    echo "   ... and $(( $(echo "$PX_FONT_SIZES" | wc -l) - 10 )) more"
  fi
  ((WARNING_COUNT++))
else
  echo "‚úÖ All font-sizes use rem"
fi

# –ü—Ä–∞–≤–∏–ª–æ 4: flex: 1; –º–∞—î –±—É—Ç–∏ flex: 1 0 0; –∞–±–æ flex: 1 0 auto;
echo ""
echo "üì¶ [Rule 4] Checking flex shorthand..."
FLEX_ISSUES=$(echo "$CSS_FILES" | xargs grep -n 'flex:\s*1;' || echo "")
if [ -n "$FLEX_ISSUES" ]; then
  echo "‚ùå Found 'flex: 1;' without explicit flex-basis:"
  echo "$FLEX_ISSUES"
  echo "   Fix: Use 'flex: 1 0 0;' or 'flex: 1 0 auto;'"
  ((ERROR_COUNT++))
else
  echo "‚úÖ All flex shorthands are explicit"
fi

# –ü—Ä–∞–≤–∏–ª–æ 5: hover –µ—Ñ–µ–∫—Ç–∏ –º–∞—é—Ç—å –±—É—Ç–∏ –≤ @media (hover: hover)
echo ""
echo "üñ±Ô∏è  [Rule 5] Checking hover effects in media query..."
HOVER_EFFECTS=$(echo "$CSS_FILES" | xargs grep -n ':hover' || echo "")
if [ -n "$HOVER_EFFECTS" ]; then
  # –ü–µ—Ä–µ–≤—ñ—Ä—è—î–º–æ, —á–∏ –≤—Å—ñ :hover –≤ @media (hover: hover)
  UNCHECKED_HOVERS=$(echo "$CSS_FILES" | while read -r file; do
    awk '
      /@media.*\(hover: hover\)/ { in_media=1; next }
      /^}/ { if (in_media) in_media=0 }
      /:hover/ { if (!in_media) print FILENAME":"NR":"$0 }
    ' "$file"
  done)
  
  if [ -n "$UNCHECKED_HOVERS" ]; then
    echo "‚ö†Ô∏è  :hover effects outside @media (hover: hover):"
    echo "$UNCHECKED_HOVERS" | head -n 5
    echo "   Recommendation: Wrap hover effects in @media (hover: hover) { ... }"
    ((WARNING_COUNT++))
  else
    echo "‚úÖ All :hover effects are in @media (hover: hover)"
  fi
else
  echo "‚úÖ No hover effects found"
fi

# –ü—Ä–∞–≤–∏–ª–æ 6: overscroll-behavior: none; –Ω–∞ body
echo ""
echo "üìú [Rule 6] Checking overscroll-behavior..."
OVERSCROLL=$(echo "$CSS_FILES" | xargs grep -c 'overscroll-behavior' | grep -v ':0$' || echo "")
if [ -z "$OVERSCROLL" ]; then
  echo "‚ö†Ô∏è  No overscroll-behavior detected"
  echo "   Recommendation: Add 'body { overscroll-behavior: none; }' to base.css"
  ((WARNING_COUNT++))
else
  echo "‚úÖ overscroll-behavior is used"
fi

# –ü—Ä–∞–≤–∏–ª–æ 7: !important –∑–∞–±–æ—Ä–æ–Ω–µ–Ω–∏–π (–¥—É–±–ª—é—î Stylelint, –∞–ª–µ –¥–ª—è –Ω–∞–¥—ñ–π–Ω–æ—Å—Ç—ñ)
echo ""
echo "üö´ [Rule 7] Checking for !important..."
IMPORTANT=$(echo "$CSS_FILES" | xargs grep -n '!important' || echo "")
```
